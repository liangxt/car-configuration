DROP DATABASE test;
CREATE DATABASE test;
USE test;

CREATE TABLE automobile(
AUTOID INTEGER NOT NULL AUTO_INCREMENT,
make VARCHAR(255),
model VARCHAR(255),
price VARCHAR(255),
PRIMARY KEY (AUTOID));

CREATE TABLE optionset(
OPSID INTEGER NOT NULL AUTO_INCREMENT,
name VARCHAR(255),
AUTOID INTEGER,
PRIMARY KEY (OPSID),
FOREIGN KEY(AUTOID) REFERENCES 
test.automobile(AUTOID) 
ON DELETE CASCADE);

CREATE TABLE optionchoice(
OPID INTEGER NOT NULL AUTO_INCREMENT,
name VARCHAR(255),
opprice VARCHAR(255),
OPSID INTEGER,
PRIMARY KEY (OPID),
FOREIGN KEY(OPSID) REFERENCES  
test.optionset(OPSID) 
ON DELETE CASCADE);
